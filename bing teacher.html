$$ \begin{aligned} \text{Proof:} \quad & \text{まず、任意の } A \in SL(2,R) \text{ と } z \in H \text{ に対して、} \alpha(A,z) \in H \text{ であることを示します。} \\ & A = \begin{pmatrix} a & b \\ c & d \end{pmatrix} \text{ とおきます。このとき、} \alpha(A,z) = \frac{az + b}{cz + d} \text{ です。} \\ & z = x + iy \text{ とすると、} \\ & \begin{aligned} \operatorname{Im}(\alpha(A,z)) &= \operatorname{Im}\left(\frac{(az + b)(c\overline{z} + d)}{|cz + d|^2}\right) \\ &= \operatorname{Im}\left(\frac{(ac|z|^2 + (ad + bc)x + bd) + i((ad - bc)y)}{|cz + d|^2}\right) \\ &= \frac{(ad - bc)y}{|cz + d|^2}. \end{aligned} \\ & \text{ここで、} ad - bc = 1 \text{ であり、} y > 0 \text{ なので、} \operatorname{Im}(\alpha(A,z)) > 0 \text{ が成り立ちます。よって、} \\ & \alpha(A,z) \in H \text{ です。} \\ & \\ & \text{次に、任意の } A,B \in SL(2,R) \text{ と } z \in H \text{ に対して、} \\ & \alpha(AB,z) = \alpha(A,\alpha(B,z)) \text{ が成り立つことを示します。} \\ & A = \begin{pmatrix} a & b \\ c & d \end{pmatrix}, B = \begin{pmatrix} e & f \\ g & h \end{pmatrix} \\ &\text {とおきます。このとき、}\\ &\begin {aligned} \alpha(AB,z) &=\alpha\left(\begin {pmatrix} a & b \\ c & d \end {pmatrix} \begin {pmatrix} e & f \\ g & h \end {pmatrix}, z\right)\\ &=\alpha\left(\begin {pmatrix} ae+bg & af+bh \\ ce+dg & cf+dh \end {pmatrix}, z\right)\\ &=\frac {(ae+bg)z+(af+bh)}{(ce+dg)z+(cf+dh)}. \end {aligned}\\ &\text {また、}\\ &\begin {aligned} \alpha(A,\alpha(B,z)) &=\alpha\left(\begin {pmatrix} a & b \\ c & d \end {pmatrix},\frac {ez+f}{gz+h}\right)\\ &=\frac {a\left(\frac {ez+f}{gz+h}\right)+b}{c\left(\frac {ez+f}{gz+h}\right)+d}\\ &=\frac {(ae+bg)z+(af+bh)}{(ce+dg)z+(cf+dh)}. \end {aligned}\\ &\text {よって、}\alpha(AB,z)=\alpha(A,\alpha(B,z))\text {が成り立ちます。}\\ &\\ &Q.E.D. \end {aligned} $$
